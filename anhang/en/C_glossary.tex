\chapter{Glossary}\label{app:glossary}

% [AP-E2/AP-EN] Mirror of anhang/de/C_glossary.tex. Abbreviations + core terms used in Chs. 1--3.
This glossary summarizes the abbreviations and central terms used in the thesis.

\begin{description}
  \item[ABI] Application Binary Interface --- the binary-stable interface through which the builder
        loads the permutation modules (living beings).
  \item[ART] Adaptive Radix Tree~\cite{leis2013art}; reference design with adaptive node sizes
        (Node4/16/48/256).
  \item[AVX, AVX-512] x86 SIMD extensions (256- and 512-bit, respectively).
  \item[Axis] An orthogonal partial decision (``organ'') of a search algorithm; the framework maintains
        nineteen composition axes.
  \item[CLU] Cache-line utilization --- the fraction of used bytes per loaded cache line.
  \item[CPI, IPC] Cycles per instruction and instructions per cycle, respectively.
  \item[CRTP] Curiously Recurring Template Pattern --- C++ idiom for static polymorphism.
  \item[dTLB] Data translation lookaside buffer; caches data address translations.
  \item[Genus] An externally visible interface (SearchAlgorithm, Container, Graph) --- the topmost level
        of the framework.
  \item[GoF] ``Gang of Four'' --- the canonical design patterns after Gamma et al.~\cite{gamma1994patterns}.
  \item[HBM] High-Bandwidth Memory.
  \item[HOT] Height Optimized Trie~\cite{binna2018hot}.
  \item[Living being] A complete search algorithm as a composition of all axes (metaphorical code term;
        technical identifier: SearchAlgorithm instance).
  \item[LOUDS] Level-Ordered Unary Degree Sequence --- succinct tree encoding (Jacobson 1989).
  \item[MESI, MOESI, MESIF] Cache coherence protocols (base protocol and AMD/Intel variants, respectively).
  \item[NINE] Non-Inclusive-Non-Exclusive --- inclusion policy of the L3 in Intel server processors.
  \item[NUMA] Non-Uniform Memory Access.
  \item[OLC] Optimistic Lock Coupling --- optimistic concurrency technique (ARTSync, P08).
  \item[PMC] (Hardware) Performance Monitoring Counter.
  \item[Probe] The new search algorithm to be integrated for testing (here PRT-ART); it is compiled by
        the builder against the core constituents of the cache engine.
  \item[Probe dock] The builder side that loads exactly one genus, drives it through its interface,
        measures it, and persists the results.
  \item[PRT-ART] Probst Redirect Tree / ART hybrid --- the probe of this thesis.
  \item[Rank] Centrality classification of the state-of-the-art publications (Rank-1/2/3); explained in
        Section~\ref{sec:rqs}.
  \item[RCU] Read-Copy-Update --- reclamation scheme (McKenney, P29).
  \item[RRIP, DRRIP] (Dynamic) Re-Reference Interval Prediction --- cache replacement
        heuristic~\cite{jaleel2010rrip}.
  \item[SIMD] Single Instruction, Multiple Data.
  \item[SOSD] Search On Sorted Data --- index benchmark~\cite{kipf2019sosd}.
  \item[SOTA] State of the Art.
  \item[SuRF] Succinct Range Filter~\cite{zhang2018surf}.
  \item[SVE, NEON, RVV] ARM (SVE, NEON) and RISC-V (RVV) SIMD extensions, respectively.
  \item[TLB] Translation Lookaside Buffer.
  \item[YCSB] Yahoo!\ Cloud Serving Benchmark~\cite{cooper2010ycsb}.
\end{description}
